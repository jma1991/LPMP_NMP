Integration

Data processing

Define function to clean SCE object:

cleanSCE <- function(x) {

  # Clear metadata
  metadata(x) <- list()

  # Replace rownames
  rownames(x) <- rowData(x)$gene_id

  # Clear rowData
  rowData(x) <- NULL

  # Replace rowRanges
  rowRanges(x) <- relist(GRanges(), PartitioningByEnd(integer(length(x)), names = rownames(x)))

  # Clear colnames
  colnames(x) <- NULL

  # Clear reducedDims
  reducedDims(x) <- NULL
  
  # Cleart altExp
  altExp(x) <- NULL

  # Return object
  return(x)

}
